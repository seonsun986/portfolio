// [GPUN] LRU 알고리즘 기반 리소스 자동 해지 풀러 구현
public class LruKeyedObjectPooler<T> {
    private readonly Dictionary<string, LinkedListNode<PoolItem>> _cache;
    private readonly LinkedList<PoolItem> _lruList;

    public T Get(string key) {
        if (_cache.TryGetValue(key, out var node)) {
            _lruList.Remove(node);
            _lruList.AddLast(node); // 최근 사용 리소스를 갱신
            return node.Value.Asset;
        }
        EvictOldestIfNeeded(); // 메모리 한계 시 가장 오래된 에셋 해지
    }
}

private void UpdateSpineHitbox(BoundingBoxAttachment box, Slot slot) {
    float[] worldVertices = new float[box.WorldVerticesLength];
    box.ComputeWorldVertices(slot, 0, box.WorldVerticesLength, worldVertices, 0, 2);
    
    Vector2[] newPoints = new Vector2[worldVertices.Length / 2];
    for (int i = 0; i < newPoints.Length; i++) {
        newPoints[i] = new Vector2(worldVertices[i*2], worldVertices[i*2+1]);
    }
    polygonCollider.SetPath(0, newPoints); // 물리 엔진 연산 없이 정밀 터치 구현
}
